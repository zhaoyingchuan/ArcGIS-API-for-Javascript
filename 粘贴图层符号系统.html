Pro中 应用图层的符号设置 工具使用起来特别费劲，构建按钮并将其插入到右键菜单中以简化步骤。

注意：并且这个操作不能被撤销和重做，事实上，使用Pro中自带的 应用图层的符号设置 工具执行这一操作也不能被撤销和重做。

protected override async void OnClick()
{
    await QueuedTask.Run(async () =>
    {
        try
        {
            var mapView = MapView.Active;

            if (mapView.GetSelectedLayers().Count() < 2)
                MessageBox.Show("运行时出现异常" + "\r\n" + "请选择两个或两个以上几何类型相同且名称不同的图层再进行操作");
            else
            {
                var sourceLayer = mapView.GetSelectedLayers()[0] as Layer;
                for (int i = 1; i< mapView.GetSelectedLayers().Count(); i++)
                {
                    var applyLayer = mapView.GetSelectedLayers()[i] as Layer;
                    // Place parameters into an array
                    var parameters = Geoprocessing.MakeValueArray(applyLayer, sourceLayer);
                    var gpResult = await Geoprocessing.ExecuteToolAsync("management.ApplySymbologyFromLayer", parameters);
                    if (gpResult.IsFailed)
                        MessageBox.Show("运行时出现异常" + "\r\n" + "请选择两个或两个以上几何类型相同且名称不同的图层再进行操作");
                }
            }
        }
        catch
        {
            // Catch any exception found and display in a message box
            MessageBox.Show("运行时出现异常" + "\r\n" + "请选择两个或两个以上几何类型相同且名称不同的图层再进行操作");
        }
    });
}
